# 1. 혹시 모를 충돌을 방지하기 위해 로컬의 main 브랜치로 이동합니다.
git checkout main

# 2. 원격 서버(origin)의 main 브랜치 데이터를 가져와 내 main 브랜치에 병합합니다.
git pull origin main

git checkout main
git pull origin main         # 로컬 main을 최신화한 뒤
git checkout 나의-작업-브랜치  # 내 작업 브랜치로 이동해서
git merge main               # 최신 main 내용을 내 브랜치에 병합
